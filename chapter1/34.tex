\section{深思牛顿-莱布尼茨公式}

\begin{theorem}
	若$f$在$[a,b]$上可积，若原函数$F$在$[a,b]$上连续、在$(a,b)$上可导，则有：
	\begin{equation}
		\int_{a}^{b}f(x)\textbf{d}x = F(b)-F(b).
	\end{equation}
\end{theorem}

\begin{proposition}
	若$f$在$[a,b]$上可积，若$F$在$[a,b]$上连续、在$(a,b)$上可导，且除有限个点外有$F'(x)=f(x)$，则有：
	\begin{equation}
		\int_{a}^{b}f(x)\textbf{d}x = F(b)-F(b).
	\end{equation}
\end{proposition}

\textbf{注意：连续函数一定可积，可导函数一定可积.}
